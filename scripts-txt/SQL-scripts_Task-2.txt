--Таблица музыкальных жанров
CREATE TABLE IF NOT EXISTS styles (
    style_id SERIAL PRIMARY KEY,
    style VARCHAR(100) NOT NULL UNIQUE CHECK (style <> '')
);

-- Таблица исполнителей
CREATE TABLE IF NOT EXISTS artists (
    artist_id SERIAL PRIMARY KEY,
    artist VARCHAR(100) NOT NULL UNIQUE CHECK (artist <> '')
);

-- Таблица альбомов
CREATE TABLE IF NOT EXISTS albums (
    album_id SERIAL PRIMARY KEY,
    album_name VARCHAR(200) NOT NULL CHECK (album_name <> ''),
    album_year INTEGER CHECK (album_year >= 1900 AND album_year <= EXTRACT(YEAR FROM CURRENT_DATE))
);

-- Таблица треков
CREATE TABLE IF NOT EXISTS tracks (
    track_id SERIAL PRIMARY KEY,
    album_id INTEGER NOT NULL REFERENCES albums(album_id),
    track_name VARCHAR(200) NOT NULL CHECK (track_name <> ''),
    duration INTEGER NOT NULL CHECK (duration > 0 AND duration <= 36000),
    UNIQUE(album_id, track_name)
);

-- Связь многие ко многим между жанрами и исполнителями
CREATE TABLE IF NOT EXISTS styles_artists (
    style_id INTEGER NOT NULL REFERENCES styles(style_id) ON DELETE CASCADE,
    artist_id INTEGER NOT NULL REFERENCES artists(artist_id) ON DELETE CASCADE,
    CONSTRAINT pk_styl_art PRIMARY KEY (style_id, artist_id)
);

-- Связь многие ко многим между исполнителями и альбомами
CREATE TABLE IF NOT EXISTS artists_albums (
    artist_id INTEGER NOT NULL REFERENCES artists(artist_id) ON DELETE CASCADE,
    album_id INTEGER NOT NULL REFERENCES albums(album_id) ON DELETE CASCADE,
    CONSTRAINT pk_arts_alb PRIMARY KEY (artist_id, album_id)
);

-- Таблица сборников
CREATE TABLE IF NOT EXISTS collections (
    collection_id SERIAL PRIMARY KEY,
    collection_name VARCHAR(200) NOT NULL CHECK (collection_name <> ''),
    collection_year INTEGER CHECK (collection_year >= 1900 AND collection_year <= EXTRACT(YEAR FROM CURRENT_DATE))
);

-- Связь многие ко многим между сборниками и треками
CREATE TABLE IF NOT EXISTS collections_tracks (
    collection_id INTEGER NOT NULL REFERENCES collections(collection_id) ON DELETE CASCADE,
    track_id INTEGER NOT NULL REFERENCES tracks(track_id) ON DELETE CASCADE,
    CONSTRAINT pk_collect_track PRIMARY KEY (collection_id, track_id)
);
